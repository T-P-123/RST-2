\section{Příklad 1E -- Průběh vnitřních sil a momentů na nosníku}

\subsection{Zadání}

Vykreslete průběh vnitřních sil a momentů po délce prutu. Vnější zatížení uvažujte jako stochastické s~normálním rozdělením, střední hodnotou dle obrázku a~variačním koeficientem $\mathrm{CoV} = 0{,}1$.

\medskip\noindent
\textbf{Schéma nosníku:}
\begin{itemize}[nosep]
    \item Podpora $A$ ($x=0$): kloubová (pin) -- reakce $V_A$, $H_A$
    \item Podpora $B$ ($x = 5\unit{m}$): válcová (roller) -- reakce $V_B$
    \item Koncentrovaný moment $M_0 = 20\unit{kN \cdot m}$ v~$x = 2\unit{m}$ (značka na obrázku, ne kloub!)
    \item Bodová síla $F = 8\unit{kN}$ v~$x = 3\unit{m}$ (směr dolů)
    \item Spojité zatížení $q = 15\unit{kN/m}$ na intervalu $x \in [5; 8]\unit{m}$
    \item Celková délka: $L = 2 + 1 + 2 + 3 = 8\unit{m}$ (převislý konec 3\,m za~$B$)
\end{itemize}

\subsection{Deterministické řešení}

\subsubsection{Statická určitost}

Nosník má 3 neznámé reakce ($V_A$, $H_A$, $V_B$) a~3 rovnice rovnováhy. Jedná se o~\textbf{prostě podepřený nosník s~převislým koncem}, staticky určitý.

\subsubsection{Výpočet reakcí}

Celková síla od spojitého zatížení:
\begin{equation}
    F_q = q \cdot L_q = 15 \cdot 3 = 45\unit{kN}, \quad x_{F_q} = 5 + \frac{3}{2} = 6{,}5\unit{m}
\end{equation}

\noindent$H_A = 0$ (žádné horizontální zatížení).

\noindent\textbf{Momentová rovnováha k~bodu $A$} ($\sum M_A = 0$, proti směru hodinových ručiček kladné):
\begin{equation}
    M_0 - F \cdot x_F + V_B \cdot x_B - F_q \cdot x_{F_q} = 0
\end{equation}
\begin{equation}
    20 - 24 + 5\,V_B - 292{,}5 = 0 \quad \Rightarrow \quad \boxed{V_B = 59{,}30\unit{kN}}
\end{equation}

\noindent\textbf{Svislá rovnováha} ($\sum F_y = 0$):
\begin{equation}
    V_A = F + F_q - V_B = 8 + 45 - 59{,}30 = \boxed{-6{,}30\unit{kN}}
\end{equation}

Záporná reakce $V_A$ znamená, že podpora $A$ táhne nosník dolů (převislý konec s~velkým zatížením ``zvedá'' levý konec).

\subsubsection{Kontrola rovnováhy}

\begin{align}
    \sum F_y &= -6{,}30 + 59{,}30 - 8 - 45 = 0 \quad \checkmark \\
    \sum M_B &= (-5) \cdot (-6{,}30) + 20 + (-2) \cdot (-8) + 1{,}5 \cdot (-45) = 0 \quad \checkmark \\
    M(x\!=\!8) &= 0 \quad \checkmark \quad \text{(volný konec)} \\
    V(x\!=\!8) &= 0 \quad \checkmark \quad \text{(volný konec)}
\end{align}

\subsubsection{Klíčové hodnoty průběhů}

Ohybový moment v~místě aplikace $M_0$ má skok $-M_0 = -20\unit{kN \cdot m}$. Celý nosník je v~oblasti záporných momentů (vzpěr/hogging):

\begin{table}[H]
\centering
\caption{Klíčové hodnoty vnitřních sil}
\begin{tabular}{lcc}
\toprule
\textbf{Veličina} & \textbf{Hodnota} & \textbf{Pozice} \\
\midrule
$V_{\max}$ & $+44{,}96\unit{kN}$ & těsně za podporou $B$ \\
$V_{\min}$ & $-14{,}30\unit{kN}$ & mezi $F$ a $B$ \\
$M_{\min}$ & $-67{,}50\unit{kN \cdot m}$ & u~podpory $B$ ($x=5$) \\
$M(2^-)$ & $-12{,}60\unit{kN \cdot m}$ & těsně před momentem \\
$M(2^+)$ & $-32{,}60\unit{kN \cdot m}$ & těsně za momentem (skok $-20$) \\
\bottomrule
\end{tabular}
\end{table}

\subsection{Stochastická analýza}

Vstupní veličiny $M_0$, $F$, $q$ mají normální rozdělení:
\begin{equation}
    M_0 \sim N(20;\; 2), \quad F \sim N(8;\; 0{,}8), \quad q \sim N(15;\; 1{,}5)
\end{equation}
kde $\sigma = \mu \cdot \mathrm{CoV} = \mu \cdot 0{,}1$.

Metodou \textbf{Monte Carlo} ($N = 100\,000$ realizací) byly získány statistiky reakcí:

\end{tabular}
\end{table}

\section{Příklad 2E -- Osový kvadratický moment L-profilu}

\subsection{Zadání}

U~tělesa podle obrázku určete osový kvadratický moment k~ose $x$ a~$y$. Střední hodnoty rozměrů jsou uvedeny na obrázku, variační koeficient je $\mathrm{CoV} = 0{,}2$. Stochastické veličiny popisujte normálním rozdělením.

\subsection{Geometrie průřezu}

Jedná se o~\textbf{L-profil (rovnoramenný úhelník)} se dvěma rameny. Počátek souřadného systému je v~levém dolním rohu profilu.

\medskip\noindent
Rozměry:
\begin{itemize}[nosep]
    \item Svislé rameno: šířka $t = 30\unit{mm}$, celková výška $h = 30 + 140 + 30 = 200\unit{mm}$
    \item Vodorovné rameno: celková šířka $b = 30 + 170 = 200\unit{mm}$, výška $t = 30\unit{mm}$
    \item Sdílený roh: $30 \times 30\unit{mm}$ (levý dolní roh)
    \item 70\,mm = hloubka 3D tělesa (extruze), neovlivňuje 2D průřez
\end{itemize}

Rozklad na 2 nepřekrývající se obdélníky:

\begin{table}[H]
\centering
\caption{Rozměry a polohy dílčích obdélníků}
\begin{tabular}{lcccc}
\toprule
\textbf{Část} & $b_i$ [mm] & $h_i$ [mm] & $x_{C,i}$ [mm] & $y_{C,i}$ [mm] \\
\midrule
Vodorovné rameno (celé) & 200 & 30 & 100{,}0 & 15{,}0 \\
Svislé rameno (nad rohem) & 30 & 170 & 15{,}0 & 115{,}0 \\
\bottomrule
\end{tabular}
\end{table}

\subsection{Deterministické řešení}

\subsubsection{Plocha průřezu}

\begin{equation}
    A = A_1 + A_2 = 200 \cdot 30 + 30 \cdot 170 = 6\,000 + 5\,100 = \boxed{11\,100\unit{mm^2}}
\end{equation}

\subsubsection{Těžiště}

\begin{equation}
    x_C = \frac{A_1 \cdot x_{C,1} + A_2 \cdot x_{C,2}}{A} = \frac{6\,000 \cdot 100 + 5\,100 \cdot 15}{11\,100} = \frac{676\,500}{11\,100} = \boxed{60{,}95\unit{mm}}
\end{equation}

\begin{equation}
    y_C = \frac{A_1 \cdot y_{C,1} + A_2 \cdot y_{C,2}}{A} = \frac{6\,000 \cdot 15 + 5\,100 \cdot 115}{11\,100} = \frac{676\,500}{11\,100} = \boxed{60{,}95\unit{mm}}
\end{equation}

\noindent\textbf{Poznámka:} $x_C = y_C$ díky symetrii rovnoramenného úhelníku ($b = h = 200\unit{mm}$).

\subsubsection{Osové kvadratické momenty (Steinerova věta)}

\begin{equation}
    I_x = \sum \left( \frac{b_i h_i^3}{12} + A_i \cdot (y_{C,i} - y_C)^2 \right)
\end{equation}

\begin{align}
    I_{x,1} &= \frac{200 \cdot 30^3}{12} + 6\,000 \cdot (15 - 60{,}95)^2 = 450\,000 + 12\,666\,180 = 13\,116\,180\unit{mm^4} \\
    I_{x,2} &= \frac{30 \cdot 170^3}{12} + 5\,100 \cdot (115 - 60{,}95)^2 = 12\,282\,500 + 14\,901\,388 = 27\,183\,888\unit{mm^4}
\end{align}

\begin{equation}
    \boxed{I_x = 13\,116\,180 + 27\,183\,888 = 40{,}30 \E{6}\unit{mm^4}}
\end{equation}

Analogicky pro $I_y$:
\begin{align}
    I_{y,1} &= \frac{200^3 \cdot 30}{12} + 6\,000 \cdot (100 - 60{,}95)^2 = 20\,000\,000 + 9\,151\,315 = 29\,151\,315\unit{mm^4} \\
    I_{y,2} &= \frac{30^3 \cdot 170}{12} + 5\,100 \cdot (15 - 60{,}95)^2 = 382\,500 + 10\,766\,253 = 11\,148\,753\unit{mm^4}
\end{align}

\begin{equation}
    \boxed{I_y = 29\,151\,315 + 11\,148\,753 = 40{,}30 \E{6}\unit{mm^4}}
\end{equation}

\noindent\textbf{Kontrola:} $I_x = I_y$, což odpovídá symetrii rovnoramenného L-profilu vůči diagonální ose.

\subsection{Stochastická analýza}

Rozměrové veličiny ($t$, přesah vodorovného ramena, výška stojiny, výška horní části) mají normální rozdělení s~$\mathrm{CoV} = 0{,}2$. Monte Carlo simulace ($N = 100\,000$):

\begin{table}[H]
\centering
\caption{Statistiky kvadratických momentů}
\begin{tabular}{lcccc}
\toprule
\textbf{Veličina} & $\boldsymbol{\mu}$ & $\boldsymbol{\sigma}$ & \textbf{CoV} & \textbf{95\% CI} \\
\midrule
$I_x$ [$\E{6}$ mm$^4$] & 42{,}73 & 20{,}78 & 0{,}486 & $[12{,}7;\; 92{,}9]$ \\
$I_y$ [$\E{6}$ mm$^4$] & 43{,}72 & 23{,}97 & 0{,}548 & --- \\
\bottomrule
\end{tabular}
\end{table}

\noindent\textbf{Poznámka:} CoV výstupních veličin ($\approx 0{,}5$) je výrazně vyšší než vstupní CoV ($0{,}2$), protože $I_x$ závisí na rozměrech ve třetí mocnině (zesílení nejistoty).

\section{Příklad 3E -- Pravděpodobnost bezporuchového provozu klínových řemenů}

\subsection{Zadání}

V~klínových řemenech byla zjištěna provozní napětí $S$ [MPa] a~meze pevnosti $R$ [MPa]:

\medskip
\noindent\textbf{Provozní napětí $S$} ($n = 10$): 223; 116; 158{,}5; 199; 136{,}5; 169{,}5; 267{,}5; 127; 184; 151

\noindent\textbf{Mez pevnosti $R$} ($n = 12$): 211; 269{,}5; 192; 301{,}5; 220{,}5; 243{,}5; 182{,}5; 257; 234; 199{,}5; 216; 205

\medskip
\noindent Jaká je pravděpodobnost bezporuchového provozu?

\subsection{Statistický popis dat}

\begin{table}[H]
\centering
\caption{Základní statistiky naměřených dat}
\begin{tabular}{lcccc}
\toprule
\textbf{Veličina} & $\boldsymbol{\bar{x}}$ [MPa] & $\boldsymbol{s}$ [MPa] & \textbf{CoV} & \textbf{Shapiro-Wilk $p$} \\
\midrule
Provozní napětí $S$ & 173{,}20 & 46{,}70 & 0{,}270 & 0{,}640 (normální $\checkmark$) \\
Mez pevnosti $R$ & 227{,}67 & 34{,}95 & 0{,}154 & 0{,}589 (normální $\checkmark$) \\
\bottomrule
\end{tabular}
\end{table}

Shapiro-Wilkův test na hladině $\alpha = 0{,}05$ potvrzuje normalitu obou souborů.

\subsection{Analytický výpočet}

Bezporuchový provoz nastává, když mez pevnosti překoná provozní napětí: $R > S$. Definujeme \textbf{rezervu spolehlivosti}:
\begin{equation}
    Z = R - S
\end{equation}

Jsou-li $R \sim N(\mu_R, \sigma_R)$ a $S \sim N(\mu_S, \sigma_S)$ nezávislé, pak:
\begin{equation}
    Z \sim N\!\left(\mu_R - \mu_S,\; \sqrt{\sigma_R^2 + \sigma_S^2}\right)
\end{equation}

\begin{align}
    \mu_Z &= 227{,}67 - 173{,}20 = 54{,}47\unit{MPa} \\
    \sigma_Z &= \sqrt{34{,}95^2 + 46{,}70^2} = 58{,}33\unit{MPa}
\end{align}

\textbf{Index spolehlivosti:}
\begin{equation}
    \beta = \frac{\mu_Z}{\sigma_Z} = \frac{54{,}47}{58{,}33} = \boxed{0{,}934}
\end{equation}

\textbf{Pravděpodobnost poruchy:}
\begin{equation}
    P_f = \Phi(-\beta) = \Phi(-0{,}934) = 0{,}1752 = 17{,}52\%
\end{equation}

\textbf{Pravděpodobnost bezporuchového provozu:}
\begin{equation}
    \boxed{P_s = 1 - P_f = 0{,}8248 = 82{,}48\%}
\end{equation}

\subsection{Monte Carlo simulace}

Simulace s~$N = 1\,000\,000$ realizací potvrzuje analytický výsledek:

\begin{table}[H]
\centering
\caption{Srovnání metod výpočtu}
\begin{tabular}{lc}
\toprule
\textbf{Metoda} & $\boldsymbol{P_s}$ [\%] \\
\midrule
Analyticky (normální) & 82{,}48 \\
Monte Carlo (normální) & 82{,}45 \\
Monte Carlo (log-normální) & 83{,}39 \\
\bottomrule
\end{tabular}
\end{table}

\caption{Příklad 3E -- Rozdělení $S$, $R$, rezerva $Z$, QQ ploty}
\label{fig:3E}
\end{figure}

\section{Příklad 4E -- MS pružnosti a deformace součásti s~vrubem}

\subsection{Zadání}

Určete pravděpodobnost dosažení mezního stavu pružnosti a~mezního stavu deformace. Proveďte citlivostní analýzu.

\begin{itemize}[nosep]
    \item $R_e = 500\unit{MPa}$, $w_{\max} = 2\unit{mm}$, $M = 100\unit{N \cdot m}$
    \item Tloušťka $t = 15\unit{mm}$
    \item Celková výška $h = 45\unit{mm}$, hloubka vrubů: $10\unit{mm}$ (horní), $6\unit{mm}$ (dolní)
    \item Poloměr zaoblení vrubu $r = 3\unit{mm}$
    \item Stochastické veličiny: log-normální rozdělení, $\mathrm{CoV} = 0{,}1$
\end{itemize}

\subsection{Deterministické řešení}

\subsubsection{Geometrie v~místě vrubu}

Čistá výška v~místě vrubu:
\begin{equation}
    h_{\mathrm{net}} = h - 10 - 6 = 45 - 16 = 29\unit{mm}
\end{equation}

Průřezový modul a~moment setrvačnosti (obdélníkový průřez):
\begin{align}
    W_{\mathrm{net}} &= \frac{t \cdot h_{\mathrm{net}}^2}{6} = \frac{15 \cdot 29^2}{6} = 2\,102{,}5\unit{mm^3} \\
    I_{\mathrm{net}} &= \frac{t \cdot h_{\mathrm{net}}^3}{12} = \frac{15 \cdot 29^3}{12} = 30\,486\unit{mm^4}
\end{align}

\subsubsection{Součinitel koncentrace napětí}

Neuberova aproximace pro vrub hloubky $a = 10\unit{mm}$ a~poloměru $r = 3\unit{mm}$:
\begin{equation}
    K_t = 1 + 2\sqrt{\frac{a}{r}} = 1 + 2\sqrt{\frac{10}{3}} = \boxed{4{,}651}
\end{equation}

\subsubsection{Napětí a průhyb}

\begin{align}
    \sigma_{\mathrm{nom}} &= \frac{M}{W_{\mathrm{net}}} = \frac{100\,000}{2\,102{,}5} = 47{,}56\unit{MPa} \\
    \sigma_{\max} &= K_t \cdot \sigma_{\mathrm{nom}} = 4{,}651 \cdot 47{,}56 = \boxed{221{,}2\unit{MPa}}
\end{align}

Součinitel bezpečnosti k~MS pružnosti:
\begin{equation}
    k = \frac{R_e}{\sigma_{\max}} = \frac{500}{221{,}2} = \boxed{2{,}26}
\end{equation}

Průhyb konzoly ($L \approx 100\unit{mm}$):
\begin{equation}
    w = \frac{M \cdot L^2}{2 \cdot E \cdot I_{\mathrm{net}}} = \frac{100\,000 \cdot 100^2}{2 \cdot 210\,000 \cdot 30\,486} = 0{,}078\unit{mm} \ll w_{\max} = 2\unit{mm}
\end{equation}

\subsection{Stochastická analýza (Monte Carlo)}

Všechny vstupní veličiny mají log-normální rozdělení s~$\mathrm{CoV} = 0{,}1$. Simulace $N = 500\,000$:

\subsubsection{MS pružnosti: $g_1 = R_e - \sigma_{\max} > 0$}

\begin{equation}
    P(\text{porucha MS pružnosti}) = P(\sigma_{\max} > R_e) = \boxed{0{,}14\%}
\end{equation}
\begin{equation}
    \beta_{\mathrm{yield}} \approx 2{,}99
\end{equation}

\subsubsection{MS deformace: $g_2 = w_{\max} - w > 0$}

\begin{equation}
    P(\text{porucha MS deformace}) \approx \boxed{0\%} \quad (\text{žádná porucha v~500\,000 realizacích})
\end{equation}

Průhyb je o~řád menší než dovolená hodnota, MS deformace tedy není kritický.

\subsection{Citlivostní analýza}

\begin{table}[H]
\centering
\caption{Faktory důležitosti (importance factors) pro MS pružnosti}
\begin{tabular}{lcc}
\toprule
\textbf{Veličina} & $\boldsymbol{\rho}(g_1, X_i)$ & $\boldsymbol{\alpha_i^2}$ [\%] \\
\midrule
$R_e$ & $+0{,}654$ & \textbf{43{,}9} \\
$h_{\mathrm{net}}$ & $+0{,}594$ & \textbf{36{,}2} \\
$M$ & $-0{,}300$ & 9{,}3 \\
$t$ & $+0{,}299$ & 9{,}2 \\
$r$ & $+0{,}119$ & 1{,}4 \\
$L$ & $+0{,}000$ & 0{,}0 \\
\bottomrule
\end{tabular}
\end{table}

\noindent\textbf{Závěr citlivostní analýzy:} Dominantní vliv má mez kluzu $R_e$ (44\%) a~výška průřezu v~místě vrubu $h_{\mathrm{net}}$ (36\%). Pro zvýšení spolehlivosti je nejúčinnější snížit variabilitu těchto dvou veličin.

\section{Příklad 5E -- Časový průběh napětí ve vrubech}

\subsection{Zadání}

Součást se dvěma zápichy, na jedné straně přivařena k~základnímu rámu, na druhé straně se ho dotýká. Je zatížena souměrně střídavou silou. Stanovte časový průběh napětí ve vrubech.

\begin{itemize}[nosep]
    \item Všechny rozměry $\pm 0{,}5\unit{mm}$
    \item $F = +1000 \div 0\unit{N}$ (pulzující zatížení)
    \item Hlavní průměr $D = 50\unit{mm}$, průměr v~zápichu $d = 35\unit{mm}$
    \item Poloměr zápichu $r = 5\unit{mm}$
    \item Délka $L = 500\unit{mm}$
    \item Předpětí $F_0 = 1{,}4 \times 10^5\unit{N}$
\end{itemize}

\subsection{Deterministické řešení}

\subsubsection{Geometrie zápichu}

\begin{align}
    A_d &= \frac{\pi d^2}{4} = \frac{\pi \cdot 35^2}{4} = 962{,}1\unit{mm^2} \\
    t_{\mathrm{depth}} &= \frac{D - d}{2} = \frac{50 - 35}{2} = 7{,}5\unit{mm}
\end{align}

\subsubsection{Součinitel koncentrace napětí}

Neuberova aproximace:
\begin{equation}
    K_t = 1 + 2\sqrt{\frac{t_{\mathrm{depth}}}{r}} = 1 + 2\sqrt{\frac{7{,}5}{5}} = \boxed{3{,}449}
\end{equation}

\subsubsection{Napětí bez předpětí}

\begin{align}
    \sigma_{\mathrm{nom,max}} &= \frac{F_{\max}}{A_d} = \frac{1000}{962{,}1} = 1{,}039\unit{MPa} \\
    \sigma_{\max} &= K_t \cdot \sigma_{\mathrm{nom,max}} = 3{,}449 \cdot 1{,}039 = \boxed{3{,}585\unit{MPa}} \\
    \sigma_{\min} &= 0\unit{MPa} \quad (\text{pulzující, } R = 0)
\end{align}

Střední napětí a~amplituda:
\begin{align}
    \sigma_m &= \frac{\sigma_{\max} + \sigma_{\min}}{2} = 1{,}793\unit{MPa} \\
    \sigma_a &= \frac{\sigma_{\max} - \sigma_{\min}}{2} = 1{,}793\unit{MPa}
\end{align}

\subsubsection{Napětí s~předpětím}

\begin{align}
    \sigma_{\max} &= K_t \cdot \frac{F_0 + F_{\max}}{A_d} = 3{,}449 \cdot \frac{141\,000}{962{,}1} = \boxed{505{,}5\unit{MPa}} \\
    \sigma_{\min} &= K_t \cdot \frac{F_0}{A_d} = 3{,}449 \cdot \frac{140\,000}{962{,}1} = 501{,}9\unit{MPa} \\
    \sigma_m &= 503{,}7\unit{MPa}, \quad \sigma_a = 1{,}79\unit{MPa}
\end{align}

\subsection{Časový průběh}

Pro pulzující zatížení $F(t) = \frac{F_{\max}}{2}\left(1 - \cos 2\pi f t\right)$:

\begin{equation}
    \sigma(t) = K_t \cdot \frac{F(t)}{A_d} = \frac{K_t \cdot F_{\max}}{2 A_d}\left(1 - \cos 2\pi f t\right)
\end{equation}

Napětí harmonicky kmitá mezi $0$ a~$\sigma_{\max}$ s~poměrem asymetrie $R = 0$.

\subsection{Stochastická analýza}

Rozměry s~tolerancí $\pm 0{,}5\unit{mm}$ $\Rightarrow$ $\sigma_{\mathrm{dim}} = 0{,}5/3 \approx 0{,}167\unit{mm}$ (pravidlo $3\sigma$).
